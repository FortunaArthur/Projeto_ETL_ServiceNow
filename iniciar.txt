# 1. Inicializa um novo projeto Node com configurações padrão
#    -y = aceita as configurações padrão automaticamente
npm init -y

# 2. Instala o TypeScript e as definições de tipo do Node.js como dependências de desenvolvimento
#    - typescript: compilador TypeScript
#    - @types/node: tipagem para APIs do Node (ex.: process, fs, etc.)
npm install --save-dev typescript @types/node

# 3. Cria o arquivo tsconfig.json com as configurações do compilador TypeScript
#    Esse arquivo define:
#       - target: versão do JavaScript de saída
#       - module: sistema de módulos (CommonJS, ESNext)
#       - rootDir e outDir: pastas de entrada e saída
#       - strict: regras de tipagem mais seguras
npx tsc --init

# 4. Instala o TSX, que é a forma mais rápida e moderna de rodar TypeScript diretamente
#    - Não precisa compilar para JS antes
#    - Suporte a ES Modules, top-level await e reload rápido
npm install --save-dev tsx
